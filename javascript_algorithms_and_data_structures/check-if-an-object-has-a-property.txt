  Jeff: {

  },
  Sarah: {

  },
  Ryan: {

function isEveryoneHere(obj) {
  // Only change code below this line
  let names = ['Alan', 'Jeff', 'Sarah', 'Ryan'];
  for (let i = 0; i < 4; i++) {
    if (!obj.hasOwnProperty(names[i])) {
      return false;
    }
  }
  return true;
  // Only change code above this line
}
